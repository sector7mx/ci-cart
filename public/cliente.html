Enviar datos en formato JSON a test.php y regresa los mismos datos JSON en un alert de jquery

Con esto se prueba elenvio y recepcion de datos tipo JSON


$(document).ready(function(){
  
    var jsondata = 
          {
              qty: 4,
              id: "123456",
              att: {
                'Linea 1': 'Algun Valor 1',
                'Linea 2': 'Algun Valor 2',
                'Linea 3': 'Algun Valor 3'
          }
    };

    $("button").click(function(){

        $.post("appartame.php",jsondata,function(data,status){

            alert("RESPUESTA JSON: " + data + "\nStatus: " + status);

        });

    });

});
